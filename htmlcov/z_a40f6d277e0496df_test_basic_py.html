1"""

3 

4These tests require the example app to be running on localhost:8080.

8"""

31class TestTextInputsComponent: 

32 """Tests for the text-inputs component with unicorn:model binding.""" 

33 

34 def test_input_updates_display(self, page: Page): 

35 """Test that typing in an input updates the displayed text.""" 

36 page.goto(f"{BASE_URL}/text-inputs", timeout=60000) 

37 

38 # Wait for component to initialize 

39 page.wait_for_selector("[unicorn\\:id]", timeout=10000) 

40 

41 # Get the first component on the page 

42 component = page.locator("[unicorn\\:id]").first 

43 

44 # The component displays "Hello {{ name|title }}" - initially "Hello World" 

45 expect(component.locator("text=Hello World")).to_be_visible() 

46 

47 # Find a text input with unicorn:model="name" that doesn't have partial updates 

48 # The first one (nth 0) has u:partial.id="model-id" which would prevent the greeting from updating 

49 text_input = component.locator("input[unicorn\\:model='name']").nth(1) 

50 expect(text_input).to_be_visible() 

51 

52 # Clear and type a new name 

53 text_input.fill("") 

54 text_input.fill("Playwright") 

55 

56 # Blur to trigger the model update 

57 text_input.blur() 

58 

59 # Wait for the server response and DOM update 

60 page.wait_for_timeout(2000) 

61 

62 # The greeting should now show "Hello Playwright" 

63 expect(component.locator("text=Hello Playwright")).to_be_visible()
